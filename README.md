// Simple animations and interactions
document.addEventListener('DOMContentLoaded', function() {
    // Animate elements on scroll
    const observerOptions = {
        threshold: 0.1,
        rootMargin: '0px 0px -50px 0px'
    };

    const observer = new IntersectionObserver(function(entries) {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                entry.target.style.opacity = '1';
                entry.target.style.transform = 'translateY(0)';
            }
        });
    }, observerOptions);

    // Observe all sections for animation
    document.querySelectorAll('.section').forEach(section => {
        section.style.opacity = '0';
        section.style.transform = 'translateY(20px)';
        section.style.transition = 'opacity 0.6s ease, transform 0.6s ease';
        observer.observe(section);
    });

    // Visitor counter simulation
    function updateVisitorCounter() {
        const counter = document.querySelector('.visitor-counter span');
        if (counter) {
            const currentCount = parseInt(counter.textContent.replace('Visitors: ', '').replace(',', ''));
            const newCount = currentCount + Math.floor(Math.random() * 5) + 1;
            counter.textContent = `Visitors: ${newCount.toLocaleString()}`;
        }
    }

    // Update counter every 30 seconds
    setInterval(updateVisitorCounter, 30000);

    // Add click effects to project cards
    document.querySelectorAll('.project-card').forEach(card => {
        card.addEventListener('click', function() {
            this.style.transform = 'scale(0.95)';
            setTimeout(() => {
                this.style.transform = '';
            }, 150);
        });
    });

    // Skill tags hover effect
    document.querySelectorAll('.skill-tag').forEach(tag => {
        tag.addEventListener('mouseenter', function() {
            this.style.background = 'linear-gradient(135deg, #764ba2, #667eea)';
        });
        tag.addEventListener('mouseleave', function() {
            this.style.background = 'linear-gradient(135deg, #667eea, #764ba2)';
        });
    });
});

// Typewriter effect for tagline
function typeWriter(element, text, speed = 50) {
    let i = 0;
    element.innerHTML = '';
    
    function type() {
        if (i < text.length) {
            element.innerHTML += text.charAt(i);
            i++;
            setTimeout(type, speed);
        }
    }
    type();
}

// Initialize typewriter effect when page loads
window.addEventListener('load', function() {
    const tagline = document.querySelector('.tagline');
    if (tagline) {
        const originalText = tagline.textContent;
        typeWriter(tagline, originalText);
    }
});
